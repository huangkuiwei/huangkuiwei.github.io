微信小程序 - 对请求接口进行二次封装
JavaScript 微信小程序
发布日期:   2018-09-12

在微信小程序中，我们用微信提供的数据请求接口 wx.request() 来接收数据，如果不进行二次封装，那每次请求数据的时候会比较麻烦，所以一般情况下会对这个接口进行二次封装。

// 请求接口统一路径
const baseURL = 'http://localhost:8080/';

function request(method, url, data, options) {
  // 完整路径
  url = baseURL + url;
  // 请求接口时是否加载 loading 动画
  if (options.lock) { 
    wx.showLoading({
      title: '加载中',
      mask: true
    })
  }
  return new Promise((resolve, reject) => {
    wx.request({
      method,
      url,
      data,
      header: options.header,
      success: res => {
        // code 为 0 时返回的数据是目标数据
        if (res.data.code === 0) { 
          resolve(res.data.datas)
        } else {
          reject(handleError(res.data))
        }
      },
      fail: () => {
        reject(handleError({
          code: -1,
          message: '接口请求失败'
        }))
      },
      complete: () => {
        // 接口请求完成之后取消 loading 动画
        wx.hideLoading()
      }
    })
  })
}

// 统一的请求错误处理函数
function handleError(error) {
  setTimeout(() => {
    if (!error.handled) {
      // 默认弹出错误信息
      wx.showToast({
        title: error.message,
        icon: 'none'
      })
    }
  }, 0);
  return error
}

// 对 post 请求进行处理
function post(url, data, options) {
  return request('post', url, data, options)
}

// 对 get 请求进行处理
function get(url, data, options) {
  return request('get', url, data, options)
}

// 将接口导出
module.exports = {
  post,
  get
};

在接下来的页面中，我们就可以使用 request 函数来请求数据了。

// 封装的接口路径
let request = require('../../util/request.js');

Page({
  data: {
    someData: []
  },
  onLoad(){
    request.post('path1XXX/path2XXX', {a: 1, b: 2}, {lock: true}).then(data => {
      this.setData({
        someData: data
      })
    })
  }
})
